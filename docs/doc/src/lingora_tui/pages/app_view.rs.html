36impl Page {
37    pub fn next(&self) -> Self {
38        let index = Page::VARIANTS.iter().position(|x| x == self).unwrap();
39        let index = (index + 1) % Page::VARIANTS.len();
40        Page::VARIANTS[index]
41    }
42
43    pub fn previous(&self) -> Self {
44        let index = Page::VARIANTS.iter().position(|x| x == self).unwrap();
45        let index = (index + Page::VARIANTS.len() - 1) % Page::VARIANTS.len();
46        Page::VARIANTS[index]
47    }
48}

59impl AppViewState {
60    pub fn new(settings: &LingoraToml, audit_result: Rc<AuditResult>) -> Self {
61        Self {
62            run_state: RunState::default(),
63            page: Page::default(),
64            translations_state: TranslationsState::new(audit_result.clone()),
65            dioxus_i18n_config_state: DioxusI18nConfigState::new(
66                settings,
67                audit_result.workspace(),
68            ),
69            settings_state: SettingsState::new(settings),
70        }
71    }
72
73    pub fn is_running(&self) -> bool {
74        matches!(self.run_state, RunState::Running)
75    }
76
77    fn handle_key_event(&mut self, event: &KeyEvent) -> Outcome {
78        match event.code {
79            KeyCode::Esc => self.quit(),
80            KeyCode::PageDown => self.set_page(self.page.next()),
81            KeyCode::PageUp => self.set_page(self.page.previous()),
82            KeyCode::F(1) => self.set_page(Page::Help),
83            _ => Outcome::Continue,
84        }
85    }
86
87    #[inline]
88    fn quit(&mut self) -> Outcome {
89        self.run_state = RunState::Quit;
90        Outcome::Changed
91    }
92
93    #[inline]
94    fn set_page(&mut self, page: Page) -> Outcome {
95        self.page = page;
96        Outcome::Changed
97    }
98
99    #[inline]
100    pub fn locale_filter(&self) -> &str {
101        self.translations_state.locale_filter()
102    }
103
104    #[inline]
105    pub fn identifier_filter(&self) -> &str {
106        self.translations_state.identifier_filter()
107    }
108}

148impl HandleEvent<Event, Regular, Outcome> for AppViewState {
149    fn handle(&mut self, event: &Event, _qualifier: Regular) -> Outcome {
150        match event {
151            Event::Key(event) => self.handle_key_event(event),
152            _ => Outcome::Continue,
153        }
154        .or_else(|| match self.page {
155            Page::Translations => self.translations_state.handle(event, Regular),
156            Page::DioxusI18nConfig => self.dioxus_i18n_config_state.handle(event, Regular),
157            Page::Settings => self.settings_state.handle(event, Regular),
158            Page::Help => Outcome::Continue,
159        })
160    }
161}

177impl<'a> StatefulWidget for &mut AppView<'a> {
178    type State = AppViewState;
179
180    fn render(self, area: Rect, buf: &mut Buffer, state: &mut Self::State) {
181        let locale_styling = &self.styling.locale;
182
183        let reference = state.translations_state.reference();
184        let target = state.translations_state.target();
185
186        let footer_style = if is_valid_footer(
187            self.audit_result,
188            &state.translations_state,
189            reference,
190            target,
191        )
192        .unwrap_or_default()
193        {
194            Style::new().light_green()
195        } else {
196            Style::new().light_red()
197        };
198
199        let node_span = |node: Option<&LocaleNode>| {
200            if let Some(node) = node {
201                match &node.kind() {
202                    LocaleNodeKind::WorkspaceRoot => Span::from("workspace"),
203                    LocaleNodeKind::LanguageRoot { language } => {
204                        locale_styling.language_root_span(language)
205                    }
206                    LocaleNodeKind::Locale { locale } => locale_styling.locale_span(locale),
207                }
208            } else {
209                Span::from("-")
210            }
211        };
212
213        let title = Line::from(vec![
214            Span::from(" Lingora - "),
215            locale_styling.locale_span(self.audit_result.canonical_locale()),
216            Span::from(" "),
217        ])
218        .centered();
219
220        let footer_left =
221            Line::from(vec![Span::from("F1").blue(), Span::from(" - Help")]).left_aligned();
222
223        let reference =
224            node_span(reference.and_then(|id| state.translations_state.locale_node(id)));
225        let target = node_span(target.and_then(|id| state.translations_state.locale_node(id)));
226
227        let footer_right = Line::from(vec![
228            Span::from("Reference: ").light_blue(),
229            reference.style(footer_style),
230            Span::from(" Target: ").light_blue(),
231            target.style(footer_style),
232            Span::from("  "),
233        ])
234        .right_aligned();
235
236        Block::new()
237            .title(title)
238            .title_bottom(footer_left)
239            .title_bottom(footer_right)
240            .render(area, buf);
241
242        let area = Rect::new(area.x + 1, area.y + 1, area.width - 2, area.height - 2);
243        match state.page {
244            Page::Translations => {
245                Translations::new(self.styling, self.audit_result).render(
246                    area,
247                    buf,
248                    &mut state.translations_state,
249                );
250            }
251            Page::DioxusI18nConfig => {
252                DioxusI18nConfig.render(area, buf, &mut state.dioxus_i18n_config_state);
253            }
254            Page::Settings => {
255                Settings.render(area, buf, &mut state.settings_state);
256            }
257            Page::Help => Help.render(area, buf),
258        };
259    }
260}

262fn is_valid_footer(
263    audit_result: &AuditResult,
264    translations_state: &TranslationsState,
265    reference: Option<&LocaleNodeId>,
266    target: Option<&LocaleNodeId>,
267) -> Option<bool> {
268    let reference_node = translations_state.locale_node(reference?)?;
269    let target_node = translations_state.locale_node(target?)?;
270
271    let (reference_locale, target_locale) = match (reference_node.kind(), target_node.kind()) {
272        (
273            LocaleNodeKind::Locale { locale: reference },
274            LocaleNodeKind::Locale { locale: target },
275        ) => (reference, target),
276        _ => return None,
277    };
278
279    let reference_doc = audit_result.document(reference_locale)?;
280    let target_doc = audit_result.document(target_locale)?;
281
282    let same_root = LanguageRoot::from(reference_locale) == LanguageRoot::from(target_locale);
283
284    let result = match (reference_doc.role(), target_doc.role()) {
285        (DocumentRole::Canonical, DocumentRole::Primary) => true,
286        (DocumentRole::Canonical | DocumentRole::Primary, DocumentRole::Variant) => same_root,
287        _ => false,
288    };
289
290    Some(result)
291}
