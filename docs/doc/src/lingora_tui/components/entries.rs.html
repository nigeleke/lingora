16impl EntriesState {
17    fn handle_key_event(&mut self, event: &KeyEvent) -> Outcome {
18        match &event.code {
19            KeyCode::Up => {
20                self.list_state.select_previous();
21                Outcome::Unchanged
22            }
23            KeyCode::Down => {
24                self.list_state.select_next();
25                Outcome::Unchanged
26            }
27            _ => Outcome::Continue,
28        }
29    }
30}
31
32impl HasFocus for EntriesState {
33    fn build(&self, builder: &mut FocusBuilder) {
34        builder.leaf_widget(self);
35    }
36
37    fn focus(&self) -> FocusFlag {
38        self.focus_flag.clone()
39    }
40
41    fn area(&self) -> Rect {
42        self.area
43    }
44}
45
46impl HandleEvent<Event, Regular, Outcome> for EntriesState {
47    fn handle(&mut self, event: &Event, _qualifier: Regular) -> Outcome {
48        if self.focus_flag.is_focused() {
49            match event {
50                Event::Key(event) => self.handle_key_event(event),
51                _ => Outcome::Continue,
52            }
53        } else {
54            Outcome::Continue
55        }
56    }
57}
