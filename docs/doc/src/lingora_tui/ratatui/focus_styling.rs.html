1use rat_focus::{FocusFlag, HasFocus};
2use ratatui::{
3    prelude::*,
4    widgets::{Block, Borders},
5};
6
7pub struct FocusStyling {
8    focused: Style,
9    unfocused: Style,
10}
11
12impl Default for FocusStyling {
13    fn default() -> Self {
14        Self {
15            focused: Style::default().light_blue(),
16            unfocused: Style::default().dim(),
17        }
18    }
19}
20
21impl FocusStyling {
22    pub fn block<'a>(&self, focus: &FocusFlag) -> Block<'a> {
23        Block::new()
24            .borders(Borders::ALL)
25            .border_style(if focus.is_focused() {
26                self.focused
27            } else {
28                self.unfocused
29            })
30    }
31}
