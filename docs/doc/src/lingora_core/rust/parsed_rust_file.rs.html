59impl TryFrom<&RustFile> for ParsedRustFile {
60    type Error = LingoraError;
61
62    fn try_from(file: &RustFile) -> Result<Self, Self::Error> {
63        let file = file.clone();
64        let source = fs::read_to_string(file.path())?;
65        let syntax = syn::parse_file(&source);
66
67        let mut macro_calls = Vec::new();
68
69        if let Ok(syntax) = &syntax {
70            let mut visitor = MacroCallVisitor::default();
71            visitor.visit_file(syntax);
72            macro_calls.extend(visitor.macro_calls);
73        };
74
75        Ok(Self {
76            file,
77            syntax,
78            macro_calls,
79        })
80    }
81}

84struct MacroCallVisitor {
85    macro_calls: Vec<MacroCall>,

88fn record_direct_macro_call(tokens: &TokenStream, macro_name: &str, out: &mut Vec<MacroCall>) {
89    if let Some(TokenTree::Literal(literal)) = tokens.clone().into_iter().next()
90        && let Ok(literal) = syn::parse2::<LitStr>(literal.into_token_stream())
91    {
92        let macro_name = String::from(macro_name);
93        let literal = literal.value();
94
95        out.push(MacroCall {
96            macro_name,
97            literal,
98        });
99    }
100}
101
102fn record_literal_macro_calls(tokens: &TokenStream, out: &mut Vec<MacroCall>) {
103    let mut iter = tokens.clone().into_iter().peekable();
104
105    while let Some(token) = iter.next() {
106        match token {
107            TokenTree::Ident(ident) if matches!(ident.to_string().as_str(), "t" | "te" | "tid") => {
108                let macro_name = ident.to_string();
109
110                if let Some(TokenTree::Punct(punct)) = iter.peek()
111                    && punct.as_char() == '!'
112                {
113                    iter.next();
114
115                    if let Some(TokenTree::Group(group)) = iter.next()
116                        && group.delimiter() == Delimiter::Parenthesis
117                    {
118                        record_direct_macro_call(&group.stream(), &macro_name, out);
119                    }
120                }
121            }
122
123            TokenTree::Group(group) => {
124                record_literal_macro_calls(&group.stream(), out);
125            }
126
127            TokenTree::Literal(literal) => {
128                if let Ok(literal) = syn::parse2::<LitStr>(literal.to_token_stream())
129                    && let Ok(tokens) = literal.value().parse::<TokenStream>()
130                {
131                    record_literal_macro_calls(&tokens, out);
132                }
133            }
134
135            _ => {}
136        }
137    }
138}

140fn handle_macro(mac: &Macro, macro_calls: &mut Vec<MacroCall>) {
141    if let Some(ident) = mac.path.get_ident() {
142        let macro_name = ident.to_string();
143
144        match macro_name.as_str() {
145            "t" | "tid" | "te" => {
146                record_direct_macro_call(&mac.tokens, &macro_name, macro_calls);
147            }
148
149            _ => record_literal_macro_calls(&mac.tokens, macro_calls),
150        }
151    }
152}

154impl<'ast> Visit<'ast> for MacroCallVisitor {
155    fn visit_stmt_macro(&mut self, node: &'ast syn::StmtMacro) {
156        handle_macro(&node.mac, &mut self.macro_calls);
157        syn::visit::visit_stmt_macro(self, node);
158    }
159
160    fn visit_item_macro(&mut self, node: &'ast syn::ItemMacro) {
161        handle_macro(&node.mac, &mut self.macro_calls);
162        syn::visit::visit_item_macro(self, node);
163    }
164
165    fn visit_expr_macro(&mut self, node: &'ast ExprMacro) {
166        handle_macro(&node.mac, &mut self.macro_calls);
167        syn::visit::visit_expr_macro(self, node);
168    }
169}
