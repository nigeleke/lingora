1use std::collections::HashSet;

3use crate::{
4    domain::{LanguageRoot, Locale},
5    fluent::FluentFile,

9#[derive(Clone, Debug)]
10pub struct Workspace {
11    fluent_files: Vec<FluentFile>,
12    canonical: Locale,
13    primaries: Vec<Locale>,
14    rust_files: Vec<RustFile>,
15}
16
17impl Workspace {
18    pub fn new(
19        fluent_files: Vec<FluentFile>,
20        canonical: Locale,
21        primaries: Vec<Locale>,
22        rust_files: Vec<RustFile>,
23    ) -> Self {
24        Self {
25            fluent_files,
26            canonical,
27            primaries,
28            rust_files,
29        }
30    }
31
32    pub fn fluent_files(&self) -> &[FluentFile] {
33        &self.fluent_files
34    }
35
36    pub fn locales(&self) -> impl Iterator<Item = &Locale> {
37        self.fluent_files
38            .iter()
39            .map(|f| f.locale())
40            .collect::<HashSet<_>>()
41            .into_iter()
42    }
43
44    pub fn language_roots(&self) -> impl Iterator<Item = LanguageRoot> {
45        self.fluent_files
46            .iter()
47            .map(|f| LanguageRoot::from(f.locale()))
48            .collect::<HashSet<_>>()
49            .into_iter()
50    }
51
52    pub fn locales_by_language_root(&self, root: &LanguageRoot) -> impl Iterator<Item = &Locale> {
53        self.fluent_files.iter().filter_map(|f| {
54            let locale = f.locale();
55            (LanguageRoot::from(locale) == *root).then_some(locale)
56        })
57    }
58
59    pub fn fluent_files_by_locale(&self, locale: &Locale) -> impl Iterator<Item = &FluentFile> {
60        let mut files = self
61            .fluent_files()
62            .iter()
63            .filter(|f| f.locale() == locale)
64            .collect::<Vec<_>>();
65        files.sort_by_key(|f| f.path());
66        files.into_iter()
67    }
68
69    pub fn canonical_locale(&self) -> &Locale {
70        &self.canonical
71    }
72
73    pub fn is_canonical_locale(&self, locale: &Locale) -> bool {
74        locale == self.canonical_locale()
75    }
76
77    pub fn primary_locales(&self) -> impl Iterator<Item = &Locale> {
78        self.primaries.iter()
79    }
80
81    pub fn is_primary_locale(&self, locale: &Locale) -> bool {
82        self.primary_locales().any(|p| p == locale)
83    }
84
85    pub fn base_locales(&self) -> impl Iterator<Item = &Locale> {
86        self.primaries
87            .iter()
88            .chain(std::iter::once(&self.canonical))
89    }
90
91    pub fn is_base_locale(&self, locale: &Locale) -> bool {
92        self.base_locales().any(|p| p == locale)
93    }
94
95    pub fn variant_locales(&self, base: &Locale) -> impl Iterator<Item = &Locale> {
96        let base_root = LanguageRoot::from(base);
97        self.fluent_files.iter().filter_map(move |f| {
98            let locale = f.locale();
99            let root = LanguageRoot::from(locale);
100            (base_root == root && base != locale).then_some(locale)
101        })
102    }
103
104    pub fn orphan_locales(&self) -> impl Iterator<Item = &Locale> {
105        let base_roots = Vec::from_iter(self.base_locales().map(LanguageRoot::from));
106        self.fluent_files.iter().filter_map(move |f| {
107            let locale = f.locale();
108            let root = LanguageRoot::from(locale);
109            (!base_roots.contains(&root)).then_some(locale)
110        })
111    }
112
113    pub fn is_orphan_locale(&self, locale: &Locale) -> bool {
114        self.orphan_locales().any(|p| p == locale)
115    }
116
117    pub fn rust_files(&self) -> &[RustFile] {
118        &self.rust_files
119    }
120}
