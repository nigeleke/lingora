11#[derive(Clone, Debug, PartialEq, Eq)]
12pub struct FluentFile {
13    path: PathBuf,
14    locale: Locale,
15}
16
17impl FluentFile {
18    pub fn path(&self) -> &Path {
19        self.path.as_path()
20    }
21
22    pub fn locale(&self) -> &Locale {
23        &self.locale
24    }
25}
26
27impl HasLocale for FluentFile {
28    fn locale(&self) -> &Locale {
29        &self.locale
30    }
31}
32
33impl TryFrom<&Path> for FluentFile {
34    type Error = LingoraError;
35
36    fn try_from(path: &Path) -> Result<Self, Self::Error> {
37        let is_fluent_ext = |path: &Path| path.extension() == Some(OsStr::new("ftl"));
38        let is_fluent_file = |path: &Path| path.is_file() && is_fluent_ext(path);
39
40        if is_fluent_file(path) {
41            let locale = Locale::try_from(path)?;
42            let path = path.to_path_buf();
43            Ok(Self { path, locale })
44        } else {
45            Err(LingoraError::InvalidFluentPath(path.to_path_buf()))
46        }
47    }
48}

50#[cfg(test)]
51mod test {

54    use tempfile::TempDir;

56    use super::*;
57
58    fn setup(folder: &str, filename: &str) -> PathBuf {
59        let mut path = TempDir::new().expect("valid temp dir").path().to_path_buf();
60
61        if !folder.trim().is_empty() {
62            path = path.join(folder);
63        };
64
65        std::fs::create_dir_all(&path).expect("valid dir");
66
67        path = path.join(filename);
68        std::fs::write(&path, "").expect("valid write");
69
70        path
71    }
72
73    #[test]
74    fn is_fluent_file_if_locale_stem_and_ftl() {
75        let locale = Locale::from_str("en-GB").expect("valid locale");
76        let path = setup("", &format!("{locale}.ftl"));
77
78        let file = FluentFile::try_from(path.as_path()).expect("valid file");
79        assert_eq!(file.locale(), &locale);
80    }
