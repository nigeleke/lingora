1use std::{collections::BTreeMap, io};
2
3use crate::{
4    audit::{AuditIssue, AuditResult, Workspace},
5    domain::Locale,
6    error::LingoraError,
7};
8
9pub struct AnalysisRenderer {
10    workspace: Workspace,
11    issues: BTreeMap<Option<Locale>, Vec<AuditIssue>>,
12}
13
14impl AnalysisRenderer {
15    pub fn new(audit_result: &AuditResult) -> Self {
16        let workspace = audit_result.workspace().clone();
17
18        let issues = audit_result.issues().fold(BTreeMap::new(), |mut acc, i| {
19            let locale = i.locale();
20            acc.entry(locale).or_insert_with(Vec::new).push(i.clone());
21            acc
22        });
23
24        Self { workspace, issues }
25    }
26
27    pub fn render<W: io::Write>(&self, out: &mut W) -> Result<(), LingoraError> {
28        self.render_workspace(out)?;
29        self.render_language(out, "Canonical:", self.workspace.canonical_locale())?;
30        self.workspace
31            .primary_locales()
32            .try_for_each(|primary| self.render_language(out, "Primary:", primary))?;
33        self.workspace
34            .orphan_locales()
35            .try_for_each(|locale| self.render_locale(out, "Orphaned:", locale))
36    }
37
38    fn render_workspace<W: io::Write>(&self, out: &mut W) -> Result<(), LingoraError> {
39        if let Some(issues) = self.issues.get(&None) {
40            let issues = issues.iter().fold(BTreeMap::new(), |mut acc, issue| {
41                acc.entry(issue.subject())
42                    .or_insert_with(Vec::new)
43                    .push(issue);
44                acc
45            });
46
47            issues.iter().try_for_each(|(subject, issues)| {
48                writeln!(out, "Workspace: {subject}")?;
49
50                issues
51                    .iter()
52                    .try_for_each(|issue| writeln!(out, "{:10} {}", "", issue.message()))
53            })?;
54        }
55
56        Ok(())
57    }
58
59    fn render_language<W: io::Write>(
60        &self,
61        out: &mut W,
62        title: &str,
63        base: &Locale,
64    ) -> Result<(), LingoraError> {
65        writeln!(out, "Language:  {}", base.language())?;
66
67        self.render_locale(out, title, base)?;
68
69        let mut variants_locales = Vec::from_iter(self.workspace.variant_locales(base));
70
71        if !variants_locales.is_empty() {
72            variants_locales.sort();
73            variants_locales
74                .iter()
75                .try_for_each(|v| self.render_locale(out, "Variant:", v))?
76        }
77
78        Ok(())
79    }
80
81    fn render_locale<W: io::Write>(
82        &self,
83        out: &mut W,
84        title: &str,
85        locale: &Locale,
86    ) -> Result<(), LingoraError> {
87        if let Some(issues) = self.issues.get(&Some(locale.clone())) {
88            writeln!(out, "{:10} {}", title, locale,)?;
89
90            let mut issues = issues.clone();
91            issues.sort_by_key(|i| i.kind().clone());
92
93            issues
94                .iter()
95                .try_for_each(|issue| writeln!(out, "{:11}{issue}", ""))?;
96        } else {
97            writeln!(out, "{:10} {} - Ok", title, locale)?;
98        }
99
100        Ok(())
101    }
102}
