14pub struct DioxusI18nConfigRenderer {
15    settings: LingoraToml,
16    workspace: Workspace,
17    target_source_path: PathBuf,
18}
19
20impl DioxusI18nConfigRenderer {
21    pub fn new(
22        settings: &LingoraToml,
23        workspace: &Workspace,
24        target_source_path: Option<&Path>,
25    ) -> Self {
26        let settings = settings.clone();
27        let workspace = workspace.clone();
28        let target_source_path = target_source_path
29            .map(|p| p.to_path_buf())
30            .unwrap_or(PathBuf::from("."));
31
32        Self {
33            settings,
34            workspace,
35            target_source_path,
36        }
37    }
38
39    pub fn render<W: io::Write>(&self, out: &mut W) -> Result<(), LingoraError> {
40        let template = r#"use dioxus_i18n::{prelude::*, *};
41use unic_langid::{langid, LanguageIdentifier};
42/*** INCLUDE ***/
43
44pub fn config(initial_language: LanguageIdentifier) -> I18nConfig {
45    I18nConfig::new(initial_language)
46/*** LOCALES ***//*** SHARES ***//*** FALLBACK ***/}
47"#
48        .replace("/*** INCLUDE ***/", &self.include())
49        .replace("/*** LOCALES ***/", &self.locales())
50        .replace("/*** SHARES ***/", &self.shares())
51        .replace("/*** FALLBACK ***/", &self.fallback());
52
53        write!(out, "{template}")?;
54        out.flush()?;
55
56        Ok(())
57    }
58
59    fn include(&self) -> String {
60        match self.settings.dioxus_i18n.config_inclusion {
61            ConfigInclusionStyle::IncludeStr => "",
62            _ => "use std::path::PathBuf;",
63        }
64        .into()
65    }
66
67    fn locales(&self) -> String {
68        match self.settings.dioxus_i18n.config_inclusion {
69            ConfigInclusionStyle::IncludeStr => self.locales_using_prefix("include_str!"),
70            ConfigInclusionStyle::PathBuf => self.locales_using_prefix("PathBuf::from"),
71            ConfigInclusionStyle::Auto => self.auto_locales(),
72        }
73    }
74
75    fn locales_using_prefix(&self, prefix: &str) -> String {
76        let mut fluent_files = Vec::from_iter(self.workspace.fluent_files());
77        fluent_files.sort_by_key(|f| f.locale());
78        fluent_files.iter().fold(String::new(), |acc, p| {
79            let locale = self.derived_locale_using_prefix(prefix, p.path());
80            format!("{}{}", acc, locale)
81        })
82    }
83
84    fn derived_locale_using_prefix(&self, prefix: &str, path: &Path) -> String {
85        Self::locale(
86            &path.file_stem().unwrap().to_string_lossy(),
87            prefix,
88            &self.relative_path_string(path),
89        )
90    }
91
92    fn relative_path_string(&self, to_maybe_relative: &Path) -> String {
93        let from = Self::to_absolute_path(&self.target_source_path);
94        let to = Self::to_absolute_path(to_maybe_relative);
95
96        let from_components = from.components().collect::<Vec<_>>();
97        let to_components = to.components().collect::<Vec<_>>();
98
99        let common_prefix_len = from_components
100            .iter()
101            .zip(&to_components)
102            .take_while(|(a, b)| a == b)
103            .count();
104
105        if common_prefix_len == 0 {
106            to_maybe_relative.to_string_lossy().to_string()
107        } else {
108            let mut result = PathBuf::new();
109            result.extend(&mut Vec::from_iter(std::iter::repeat_n(
110                "..",
111                from_components[common_prefix_len..].len(),
112            )));
113            result.extend(&to_components[common_prefix_len..]);
114            result.to_string_lossy().to_string()
115        }
116    }
117
118    fn to_absolute_path(path: &Path) -> PathBuf {
119        if path.is_absolute() {
120            path.to_path_buf()
121        } else {
122            std::env::current_dir().unwrap().join(path)
123        }
124    }
125
126    fn locale(langid: &str, prefix: &str, path_str: &str) -> String {
127        format!(
128            r#"        .with_locale((
129            langid!("{}"),
130            {}
131        ))
132"#,
133            langid,
134            Self::locale_prefix_pathstr(prefix, path_str)
135        )
136    }
137
138    fn locale_prefix_pathstr(prefix: &str, path_str: &str) -> String {
139        format!("{}(\"{}\")", prefix, path_str)
140    }
141
142    fn auto_locales(&self) -> String {
143        self.settings
144            .lingora
145            .fluent_sources
146            .iter()
147            .filter(|source| source.is_dir())
148            .fold(String::new(), |acc, path| {
149                format!(
150                    r#"{}        .with_auto_locales(PathBuf::from("{}"))
151"#,
152                    acc,
153                    self.relative_path_string(path)
154                )
155            })
156    }
157
158    fn shares(&self) -> String {
159        match self.settings.dioxus_i18n.config_inclusion {
160            ConfigInclusionStyle::IncludeStr => self.shares_using_prefix("include_str!"),
161            ConfigInclusionStyle::PathBuf => self.shares_using_prefix("PathBuf::from"),
162            ConfigInclusionStyle::Auto => self.shares_using_prefix("PathBuf::from"),
163        }
164    }
165
166    fn shares_using_prefix(&self, prefix: &str) -> String {
167        let base_locales = self.workspace.base_locales().collect::<HashSet<_>>();
168        let language_root_only = base_locales
169            .iter()
170            .filter(|l| l.region().is_none() && !l.has_variants())
171            .collect::<HashSet<_>>();
172        let unrooted_base_files = self.workspace.fluent_files().iter().filter(|f| {
173            let locale = f.locale();
174            base_locales.contains(locale) && !language_root_only.contains(&locale)
175        });
176
177        unrooted_base_files.fold(String::new(), |acc, file| {
178            let locale = file.locale();
179            format!(
180                r#"{}        .share(langid!("{}"), {}("{}"))
181            "#,
182                acc,
183                LanguageRoot::from(locale),
184                prefix,
185                self.relative_path_string(file.path())
186            )
187        })
188    }
189
190    fn fallback(&self) -> String {
191        format!(
192            r#"        .with_fallback(langid!("{}"))
193"#,
194            self.settings.lingora.canonical
195        )
196    }
197}
