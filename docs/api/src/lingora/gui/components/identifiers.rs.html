84#[component]
85fn IdentifiersTree(filter: Signal<Filter>) -> Element {
86    let mut state = use_context::<Signal<State>>();
87
88    let default_all = !(filter.read().ok || filter.read().warnings || filter.read().errors);
89
90    let mut reference_file = use_signal(|| Ok(FluentFile::default()));
91    use_effect(move || reference_file.set(FluentFile::try_from(state.read().reference_path())));
92
93    let mut reference_identifiers = use_signal(HashSet::new);
94    use_effect(move || match &*reference_file.read() {
95        Ok(file) => reference_identifiers.set(file.identifiers()),
96        Err(e) => state.write().set_error_string(&e.to_string()),
97    });
98
99    let mut target_file = use_signal(|| None);
100    use_effect(move || target_file.set(state.read().target_path().map(FluentFile::try_from)));
101
102    let mut target_identifiers = use_signal(HashSet::new);
103    use_effect(move || {
104        if let Some(file) = &*target_file.read() {
105            match file {
106                Ok(file) => target_identifiers.set(file.identifiers()),
107                Err(e) => state.write().set_error_string(&e.to_string()),
108            }
109        }
110    });
111
112    let mut identifiers = use_signal(Vec::new);
113    use_effect(move || {
114        let all_identifiers = reference_identifiers()
115            .into_iter()
116            .chain(target_identifiers().into_iter())
117            .filter(|id| {
118                id.path()
119                    .to_ascii_lowercase()
120                    .contains(&filter.read().name.to_ascii_lowercase())
121            })
122            .filter(|id| {
123                match (
124                    reference_identifiers.read().contains(id),
125                    target_identifiers.read().contains(id),
126                ) {
127                    (true, true) => filter.read().ok || default_all,
128                    (true, false) => filter.read().errors || default_all,
129                    (false, true) => filter.read().warnings || default_all,
130                    (false, false) => unreachable!(),
131                }
132            })
133            .collect::<HashSet<_>>();
134
135        let mut sorted = Vec::from_iter(all_identifiers);
136        sorted.sort();
137
138        identifiers.set(sorted);
139    });
140
141    rsx! {
142        div {
143            class: "identifiers-tree",
144            ul {
145                for identifier in &*identifiers.read() {
146                    IdentifierItem {
147                        identifier: identifier.clone(),
148                        in_reference: reference_identifiers.read().contains(identifier),
149                        in_target: target_identifiers.read().contains(identifier)
150                    }
151                }
152            }
153        }
154    }
155}

157#[component]
158fn IdentifierItem(identifier: Identifier, in_reference: bool, in_target: bool) -> Element {
159    let mut state = use_context::<Signal<State>>();
160
161    let css_class = match (in_reference, in_target) {
162        (false, false) => "",
163        (false, true) => "superfluous-target",
164        (true, false) => "missing-target",
165        (true, true) => "both",
166    };
167
168    let select_identifier = |identifier: &Identifier| {
169        let identifier = identifier.clone();
170        move |_| {
171            state.write().set_identifier(&identifier);
172        }
173    };
174
175    let select_identifier_on_enter = |identifier: &Identifier| {
176        let identifier = identifier.clone();
177        move |event: Event<KeyboardData>| {
178            if event.data.key() == Key::Enter {
179                state.write().set_identifier(&identifier);
180            }
181        }
182    };
183
184    rsx! {
185        li {
186            class: "{css_class}",
187            class: if Some(&identifier) == state.read().identifier() { "selected" },
188            tabindex: "0",
189            role: "button",
190            key: "{identifier.path()}",
191            onclick: select_identifier(&identifier),
192            onkeypress: select_identifier_on_enter(&identifier),
193            div {
194                span { {identifier.path()} }
195                span { }
196            }
197        }
198
199    }
