43#[component]
44fn IdentifiersList(filter: Signal<String>) -> Element {
45    let mut state = use_context::<Signal<State>>();
46
47    let mut reference_file = use_signal(|| Ok(FluentFile::default()));
48    use_effect(move || reference_file.set(FluentFile::try_from(state.read().reference_path())));
49
50    let mut reference_identifiers = use_signal(HashSet::new);
51    use_effect(move || match &*reference_file.read() {
52        Ok(file) => reference_identifiers.set(file.identifiers()),
53        Err(e) => state.write().set_error_string(&e.to_string()),
54    });
55
56    let mut target_file = use_signal(|| None);
57    use_effect(move || target_file.set(state.read().target_path().map(FluentFile::try_from)));
58
59    let mut target_identifiers = use_signal(HashSet::new);
60    use_effect(move || {
61        if let Some(file) = &*target_file.read() {
62            match file {
63                Ok(file) => target_identifiers.set(file.identifiers()),
64                Err(e) => state.write().set_error_string(&e.to_string()),
65            }
66        }
67    });
68
69    let mut identifiers = use_signal(Vec::new);
70    use_effect(move || {
71        let all_identifiers = reference_identifiers()
72            .into_iter()
73            .chain(target_identifiers().into_iter())
74            .filter(|id| {
75                id.name()
76                    .to_ascii_lowercase()
77                    .contains(&filter.read().to_ascii_lowercase())
78            })
79            .collect::<HashSet<_>>();
80
81        let mut sorted = Vec::from_iter(all_identifiers);
82        sorted.sort();
83
84        identifiers.set(sorted);
85    });
86
87    rsx! {
88        div {
89            class: "identifiers-list",
90            ul {
91                for identifier in &*identifiers.read() {
92                    IdentifierItem {
93                        identifier: identifier.clone(),
94                        in_reference: reference_identifiers.read().contains(identifier),
95                        in_target: target_identifiers.read().contains(identifier)
96                    }
97                }
98            }
99        }
100
101    }
102}

104#[component]
105fn IdentifierItem(identifier: Identifier, in_reference: bool, in_target: bool) -> Element {
106    let mut state = use_context::<Signal<State>>();
107
108    let css_class = match (in_reference, in_target) {
109        (false, false) => "",
110        (false, true) => "superfluous-target",
111        (true, false) => "missing-target",
112        (true, true) => "both",
113    };
114
115    let select_identifier = |identifier: &Identifier| {
116        let identifier = identifier.clone();
117        move |_| {
118            state.write().set_identifier(&identifier);
119        }
120    };
121
122    let select_identifier_on_enter = |identifier: &Identifier| {
123        let identifier = identifier.clone();
124        move |event: Event<KeyboardData>| {
125            if event.data.key() == Key::Enter {
126                state.write().set_identifier(&identifier);
127            }
128        }
129    };
130
131    rsx! {
132        li {
133            class: "{css_class}",
134            class: if Some(&identifier) == state.read().identifier() { "selected" },
135            tabindex: "0",
136            role: "button",
137            key: "{identifier.name()}",
138            onclick: select_identifier(&identifier),
139            onkeypress: select_identifier_on_enter(&identifier),
140            div {
141                span { {identifier.name()} }
142                span { }
143            }
144        }
145
146    }
147}
