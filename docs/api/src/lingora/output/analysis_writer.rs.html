1use std::path::PathBuf;

3use super::{

9pub struct AnalysisWriter<'a> {
10    reference: &'a PathBuf,
11    analysis: &'a Analysis,
12    writer: Writer,
13}
14
15impl<'a> AnalysisWriter<'a> {
16    pub fn new(reference: &'a PathBuf, analysis: &'a Analysis, writer: Writer) -> Self {
17        Self {
18            reference,
19            analysis,
20            writer,
21        }
22    }
23
24    pub fn write(&self) -> Result<()> {
25        self.output_check("Reference:", self.reference)?;
26
27        let mut paths = self
28            .analysis
29            .paths()
30            .into_iter()
31            .filter(|p| p != &self.reference)
32            .collect::<Vec<_>>();
33        paths.sort();
34
35        paths
36            .iter()
37            .try_for_each(|f| self.output_check("Target:", f))
38    }
39
40    pub fn output_check(&self, title: &str, path: &PathBuf) -> Result<()> {
41        let path_string = path.to_string_lossy();
42        let mut checks = self.analysis.checks(path).clone();
43        checks.sort();
44
45        let mut stdout = (*self.writer).borrow_mut();
46
47        writeln!(
48            stdout,
49            "{} {}{}",
50            title,
51            path_string,
52            if checks.is_empty() { " - Ok" } else { "" }
53        )
54        .map_err(|e| WriterError::WriteFailed(e.to_string()))?;
55
56        let mut current_category = "";
57        checks
58            .iter()
59            .try_for_each(|c| {
60                if current_category != c.category_str() {
61                    current_category = c.category_str();
62                    writeln!(stdout, "    {}", c)
63                } else {
64                    writeln!(
65                        stdout,
66                        "    {}  {}",
67                        " ".repeat(c.category_str().len()),
68                        c.value_str()
69                    )
70                }
71            })
72            .map_err(|e| WriterError::WriteFailed(e.to_string()))?;
73
74        Ok(())
75    }
76}
